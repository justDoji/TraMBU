package be.doji.productivity.trambu.timetracking.domain;

import static be.doji.productivity.trambu.timetracking.domain.TimeNotation.date;
import static be.doji.productivity.trambu.timetracking.domain.TimeNotation.dateTime;
import static be.doji.productivity.trambu.timetracking.domain.TimeNotation.dateTimeExtended;
import static be.doji.productivity.trambu.timetracking.domain.TimeNotation.legacy;

import com.google.re2j.Pattern;
import java.time.Clock;
import java.time.LocalDate;
import java.time.LocalDateTime;
import java.time.ZoneId;
import java.time.format.DateTimeFormatter;
import java.util.Collections;
import java.util.Date;
import java.util.HashMap;
import java.util.Map;
import java.util.Map.Entry;
import java.util.Optional;
import java.util.TimeZone;
import org.apache.commons.lang3.StringUtils;

class TimePointFactory {

  private static Clock clock = Clock.systemDefaultZone();

  private static final Map<Pattern, DateTimeFormatter> DATE_CONVERTERS = createDateConverters();
  private static final Map<Pattern, DateTimeFormatter> DATE_TIME_CONVERTERS = createDateTimeConverters();


  private static Map<Pattern, DateTimeFormatter> createDateConverters() {
    return Collections.singletonMap(Pattern.compile(date().pattern()), date().formatter());
  }

  private static Map<Pattern, DateTimeFormatter> createDateTimeConverters() {
    Map<Pattern, DateTimeFormatter> converters = new HashMap<>();
    converters.put(Pattern.compile(dateTime().pattern()), dateTime().formatter());
    converters.put(Pattern.compile(dateTimeExtended().pattern()), dateTimeExtended().formatter());
    converters.put(Pattern.compile(legacy().pattern()), legacy().formatter());
    return converters;
  }

  public static TimePoint now() {
    return new TimePoint(LocalDateTime.ofInstant(clock.instant(), clock.getZone()));
  }

  public static TimePoint fromString(String timeString) {
    throwExceptionIfBlank(timeString);

    Optional<TimePoint> timePoint = convertToDateTime(timeString);
    if (timePoint.isPresent()) {return timePoint.get();}

    Optional<TimePoint> timePoint1 = convertToDate(timeString);
    if (timePoint1.isPresent()) {return timePoint1.get();}

    throw couldNotParseException(timeString);
  }

  private static Optional<TimePoint> convertToDateTime(String timeString) {
    for (Entry<Pattern, DateTimeFormatter> entry : DATE_TIME_CONVERTERS.entrySet()) {
      if (entry.getKey().matcher(timeString).matches()) {
        LocalDateTime parsedDateTime = LocalDateTime.parse(timeString, entry.getValue());
        return Optional.of(new TimePoint(parsedDateTime));
      }
    }
    return Optional.empty();
  }

  private static Optional<TimePoint> convertToDate(String timeString) {
    for (Entry<Pattern, DateTimeFormatter> entry : DATE_CONVERTERS.entrySet()) {
      if (entry.getKey().matcher(timeString).matches()) {
        LocalDateTime persedDate = LocalDate.parse(timeString, entry.getValue()).atStartOfDay();
        return Optional.of(new TimePoint(persedDate));
      }
    }
    return Optional.empty();
  }

  private static void throwExceptionIfBlank(String timeString) {
    if (StringUtils.isBlank(timeString)) {
      throw new IllegalArgumentException(
          "Could not parse given Date string: No matching parsers found for string [" + timeString
              + "] ");
    }
  }

  private static IllegalArgumentException couldNotParseException(String timeString) {
    return new IllegalArgumentException(
        String.format(
            "Could not parse given Date string: No matching parsers found for string [%s] ",
            timeString)
    );
  }

  public static TimePoint fromDateTime(LocalDateTime date) {
    if (date == null) {
      return null;
    }
    return new TimePoint(date);
  }

  public static TimePoint fromDate(Date date) {
    if (date == null) {
      return null;
    }
    LocalDateTime localDateTime = LocalDateTime
        .ofInstant(date.toInstant(), ZoneId.of(TimeZone.getDefault().getID()));
    return new TimePoint(localDateTime);
  }

  /**
   * static setter for testing purposes
   */
  public static void setTimePointClock(Clock toSet) {
    clock = toSet;
  }
}
