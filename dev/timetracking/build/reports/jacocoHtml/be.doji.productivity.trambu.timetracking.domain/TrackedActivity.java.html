package be.doji.productivity.trambu.timetracking.domain;

import java.util.UUID;

public class TrackedActivity {

  private final String title;
  private final UUID correlationId;

  private TrackedActivity(String title, UUID correlationId) {
    this.title = title;
    this.correlationId = correlationId;
  }

  public static Builder builder() {
    return new Builder();
  }

  public String getTitle() {
    return title;
  }

  public static class Builder {

    private String title;
    private UUID correlationId;

    public Builder() {
    }

    public TrackedActivity build() {
      return new TrackedActivity(this.title, correlationId);
    }

    public Builder title(String title) {
      this.title = title;
      return this;
    }

    public Builder correlationId(UUID correlationId) {
      this.correlationId = correlationId;
      return this;
    }
  }
}
