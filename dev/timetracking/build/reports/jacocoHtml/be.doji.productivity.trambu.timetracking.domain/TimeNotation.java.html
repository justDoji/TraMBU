package be.doji.productivity.trambu.timetracking.domain;

import java.time.format.DateTimeFormatter;
import java.util.Locale;

public class TimeNotation {

  private final String regex;
  private final String format;

  private TimeNotation(String regex, String format) {
    this.regex = regex;
    this.format = format;
  }

  public String pattern() {
    return regex;
  }

  public String format() {
    return format;
  }

  public DateTimeFormatter formatter() {
    return DateTimeFormatter.ofPattern(this.format(), Locale.FRANCE);
  }

  private static final String BASIC_DATE_REGEX = "\\d\\d/\\d\\d/\\d\\d\\d\\d";
  private static final String BASIC_DATE_PATTERN = "dd/MM/uuuu";

  public static TimeNotation date() {
    return new TimeNotation(BASIC_DATE_REGEX, BASIC_DATE_PATTERN);
  }

  private static final String BASIC_DATE_TIME_PATTERN = "dd/MM/uuuu HH:mm:ss";
  private static final String BASIC_DATE_TIME_REGEX = "\\d\\d/\\d\\d/\\d\\d\\d\\d \\d\\d:\\d\\d:\\d\\d";

  public static TimeNotation dateTime() {
    return new TimeNotation(BASIC_DATE_TIME_REGEX, BASIC_DATE_TIME_PATTERN);
  }

  private static final String EXTENDED_DATE_TIME_PATTERN = "dd/MM/yyyy HH:mm:ss:SSS";
  private static final String EXTENDED_DATE_TIME_REGEX = "\\d\\d/\\d\\d/\\d\\d\\d\\d \\d\\d:\\d\\d:\\d\\d:\\d\\d\\d";

  public static TimeNotation dateTimeExtended() {
    return new TimeNotation(EXTENDED_DATE_TIME_REGEX, EXTENDED_DATE_TIME_PATTERN);
  }


  private static final String LEGACY_DATE_TIME_PATTERN = "uuuu-MM-dd:HH:mm:ss.SSS";
  private static final String LEGACY_DATE_TIME_REGEX = "\\d\\d\\d\\d-\\d\\d-\\d\\d:\\d\\d:\\d\\d:\\d\\d.\\d\\d\\d";

  public static TimeNotation legacy() {
    return new TimeNotation(LEGACY_DATE_TIME_REGEX, LEGACY_DATE_TIME_PATTERN);
  }


}
